Basics

Y.all('code')
Y.all('#myid')
Y.all('.myclass')
Y.all('*')
Y.all('code, #myid, .myclass')

Hierarchy

Y.all('div code')
Y.all('li > ul')
Y.all('strong + em')
Y.all('strong ~ em')
